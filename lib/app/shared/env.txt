class Env {
  static (bool, String) get userDemoEmail => _get('USER_DEMO_EMAIL');
  static (bool, String) get userDemoPassword => _get('USER_DEMO_PASS');
  static (bool, String) get backendUrl => _get('BACKEND_URL');

  static (bool, String) _get(String key) {
    final value = dotenv.get(key, fallback: 'NOT_FOUND');

    switch (value) {
      case 'NOT_FOUND':
        return (true, '$key not found');
      case '':
        return (true, '$key is empty');
      default:
        return (false, value);
    }
  }
}
